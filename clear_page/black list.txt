function iterateBlocks()
{
	
	var ahref = document.getElementsByTagName("A");
	
	for(i = 0; i < ahref.length; i++)
	{
		if(ahref[i].innerHTML == "Удалить из списка")
		{
			ahref[i].onclick();
		}
	}
}

iterateBlocks();
